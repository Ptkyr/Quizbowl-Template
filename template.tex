%%-----------Quizbowl Commands------------
\newenvironment{tossuplist}
{\begin{enumerate}[wide=0pt, 
listparindent=1.25em, 
parsep=0pt, 
itemsep=10pt]}
{\end{enumerate}}
\newcommand{\answerline}[1]
{\par \noindent ANSWER: #1}
\newcommand{\need}[1]{\ul{\textbf{#1}}} % Necessary answer line
\newcommand{\category}[1]{\par\noindent \textless #1\textgreater}
\newcommand{\pguide}[1]{\textcolor{pguidecol}{(``\small#1")}} % pronunciation guide
\newcommand{\quotes}[1]{``#1"}
\newcommand{\prompt}[1]{\ul{#1}}
% toss up: optional category, mandatory clue, answer line
\NewDocumentCommand{\tossup}{omm}
{\item \vbox{#2 % vbox ensures a tossup will not be split across pages
\answerline{#3}
\IfNoValueTF{#1}
    {}
    {\category{#1}}%
}}
